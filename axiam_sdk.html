axiam_sdk

Official Python client SDK for AXIAM (Access eXtended Identity and Authorization Management).

Re-exports the public surface: the sync AxiamClient and async AsyncAxiamClient REST entry points, the AuthError/AuthzError/ NetworkError exception taxonomy (CONTRACT.md §2), and the request/response models (LoginResult, User, AccessCheck, AccessResult, BatchCheckResult). See CONTRACT.md §1-§10 for the cross-language behavioral contract this SDK conforms to.

This module MUST remain importable with ONLY the runtime dependencies declared in [project.dependencies] (httpx, grpcio, aio-pika, pydantic, PyJWT) — the optional web-framework integrations (axiam_sdk.fastapi, axiam_sdk.django, see [project.optional-dependencies]) MUST NOT be imported from here.

388    def refresh(self) -> None:
389        """``POST /api/v1/auth/refresh`` (CONTRACT.md §1), routed through
390        the shared single-flight guard (§9) so concurrent 401s collapse
391        into exactly one in-flight refresh call. A 401 on the refresh call
392        itself is ``AuthError`` with no retry (§9.3, Pitfall 4)."""
393        observed_access = self._session.cookie_value(ACCESS_COOKIE)
394        if not observed_access:

398        # D-15: diagnostic-only, never a token value. Off by default
399        # (NullHandler); integrates with the consuming app's logging config.
400        self._logger.debug("axiam_sdk: token refresh triggered")
401        self._session.refresh_guard.refresh_if_needed_sync(
402            observed_access, lambda: self._do_refresh_sync(tenant_id, org_id)

411        one in-flight call (§9)."""

413        # Path-scoped axiam_refresh cookie attaches (Pitfall 4).

457    def _authz_post_sync(self, path: str, body: dict[str, Any]) -> dict[str, Any]:
458        """POST an authz request body to *path*, transparently retrying once
459        via :meth:`_retry_after_refresh_sync` on a 401 (§9.3), and returning
460        the parsed JSON response body. Raises the mapped ``AxiamError``
461        family exception (CONTRACT.md §2) for any other non-2xx status."""
462        request = self._session.sync_client.build_request("POST", path, json=body)
463        response = self._session._send_sync(request)
464
465        if response.status_code == httpx.codes.UNAUTHORIZED:
466            response = self._retry_after_refresh_sync(request)
467
468        if response.status_code < 200 or response.status_code >= 300:
469            raise error_from_http_status(
470                response.status_code, "authz check failed", response=response
471            )
472        result: dict[str, Any] = response.json()
473        return result
474
475    def _retry_after_refresh_sync(self, original_request: httpx.Request) -> httpx.Response:
476        """On a 401, refresh exactly once (via the shared single-flight
477        guard) then retry the failed authz call exactly once. A second
478        failure propagates through the caller's own status check (§9.3, no
479        retry loop)."""
480        self.refresh()
481        retry_request = self._session.sync_client.build_request(
482            original_request.method,
483            original_request.url,
484            content=original_request.content,
485            headers={

487                for k, v in original_request.headers.items()
488                if k.lower() not in ("content-length", "x-csrf-token")
489            },
490        )
491        return self._session._send_sync(retry_request)
