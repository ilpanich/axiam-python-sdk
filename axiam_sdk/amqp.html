axiam_sdk.amqp

AXIAM SDK AMQP transport (CONTRACT.md §8, D-02).

Async-only event consumer (aio-pika) with mandatory HMAC-SHA256 verify-before-handler and v2 replay protection (NEW-4). Public surface:

  • consume() — closure-handler consumer; the SDK owns the ack/nack loop, verifies every delivery's HMAC signature before the caller's handler is ever invoked, and rejects stale/replayed/pre-v2 messages.
  • ErrDrop — sentinel a handler raises to signal "poison message, nack WITHOUT requeue" (mirrors Go's exported ErrDrop).
  • verify_hmac() — the underlying HMAC-SHA256 verifier (19-01), re-exported for callers that need to verify a message body directly.
  • NonceStore / validate_freshness() / DEFAULT_SKEW_SECONDS — the NEW-4 replay-protection primitives consume() uses internally, re-exported for callers that need to verify freshness directly or inject a shared store across multiple consumers.

182

64class ErrDrop(Exception):
65    """Sentinel a handler raises to signal "poison message" (D-02, §8).
66
67    Raising ``ErrDrop`` from a ``consume()`` handler causes the delivery to

def verify_hmac(signing_key: bytes, body: bytes) -> bool: 
30def verify_hmac(signing_key: bytes, body: bytes) -> bool:
31    """Verify an HMAC-SHA256 signature over an AMQP message body.
32
33    Returns ``True`` only if the message's ``hmac_signature`` field
34    (hex-encoded) matches HMAC-SHA256(signing_key, canonical_json) where
35    ``canonical_json`` is the message with ``hmac_signature`` removed,
36    re-serialized via ``json.dumps(msg, separators=(",", ":"))`` with the
37    keys left in their received (insertion) order — preserving the wire
38    order emitted by the server's declared-order struct serialization.
39
40    Never raises. Returns ``False`` for: malformed JSON, a body that does
41    not decode to a JSON object, a missing ``hmac_signature`` field, a
42    non-hex or wrong-length signature, a tampered body, or a wrong signing
43    key. The signature and computed digest are never logged or returned —
44    only this boolean (T-19-03).
45    """
46    try:
47        msg = json.loads(body)
48    except (json.JSONDecodeError, UnicodeDecodeError, TypeError):
49        return False
50
51    if not isinstance(msg, dict):
52        return False
53
54    sig_hex = msg.pop("hmac_signature", None)
55    if sig_hex is None or not isinstance(sig_hex, str):
56        # Strict mode default (CONTRACT.md §8.3): a message with no
57        # hmac_signature field (or a non-string value) is rejected, never
58        # silently accepted.
59        return False
60
61    # Keys are left in their received order (see module docstring) — do not
62    # alphabetize them. This reproduces the exact byte sequence the Rust
63    # signer actually signed.
64    canonical = json.dumps(msg, separators=(",", ":")).encode("utf-8")
65
66    try:
67        expected = bytes.fromhex(sig_hex)
68    except ValueError:
69        return False
70
71    computed = hmac.new(signing_key, canonical, hashlib.sha256).digest()
72    return hmac.compare_digest(computed, expected)

class NonceStore: 
45class NonceStore:
46    """In-memory nonce dedup store, naturally bounded by TTL pruning.
47
48    Maps ``nonce -> expiry`` (a :func:`time.monotonic` timestamp). A nonce
49    is recorded the first time it is seen; any repeat before its entry
50    expires is a replay and is rejected. Pruning is opportunistic — it runs
51    on every :meth:`check_and_record` call rather than on a timer/background
52    task — so the store never needs its own scheduling and stays roughly
53    bounded to (messages seen within one TTL window) entries.
54
55    Exactly ONE instance MUST be shared across every delivery on a given
56    consumer loop: :func:`axiam_sdk.amqp._consumer.consume` creates a
57    single store and threads it through every call to ``_on_message`` for
58    the lifetime of the consumer. A fresh store per-message (or per-call)
59    would make replay detection a no-op, since nothing would ever be
60    remembered between deliveries.
61    """
62
63    def __init__(self, ttl_seconds: float) -> None:
64        """Build an empty store with entries expiring ``ttl_seconds`` after
65        they are recorded. Callers should size this to cover the freshness
66        window with margin — :func:`consume` uses ``2 * skew_seconds``."""
67        self._ttl_seconds = ttl_seconds
68        self._seen: dict[str, float] = {}
69
70    def check_and_record(self, nonce: str, *, now: float | None = None) -> bool:
71        """Record ``nonce`` and report whether it was fresh.
72
73        Returns ``True`` the first time a given ``nonce`` is seen (and
74        records it with an expiry of ``now + ttl_seconds``). Returns
75        ``False`` if ``nonce`` is already present and unexpired — the
76        caller MUST treat that as a replay and reject the message.
77        """
78        if now is None:
79            now = time.monotonic()
80        self._prune(now)
81        if nonce in self._seen:
82            return False
83        self._seen[nonce] = now + self._ttl_seconds
84        return True
85
86    def _prune(self, now: float) -> None:
87        """Drop every entry whose expiry is at or before ``now``. Called
88        opportunistically at the start of each :meth:`check_and_record`
89        rather than on a timer, so the store needs no background
90        scheduling."""
91        expired = [n for n, expiry in self._seen.items() if expiry <= now]
92        for n in expired:
93            del self._seen[n]
94
95    def __len__(self) -> int:
96        """The number of currently unexpired nonces on record (does not
97        trigger pruning itself)."""
98        return len(self._seen)

In-memory nonce dedup store, naturally bounded by TTL pruning.

Maps nonce -> expiry (a time.monotonic() timestamp). A nonce is recorded the first time it is seen; any repeat before its entry expires is a replay and is rejected. Pruning is opportunistic — it runs on every check_and_record() call rather than on a timer/background task — so the store never needs its own scheduling and stays roughly bounded to (messages seen within one TTL window) entries.

Exactly ONE instance MUST be shared across every delivery on a given consumer loop: axiam_sdk.amqp._consumer.consume() creates a single store and threads it through every call to _on_message for the lifetime of the consumer. A fresh store per-message (or per-call) would make replay detection a no-op, since nothing would ever be remembered between deliveries.

def validate_freshness( event: dict[str, typing.Any], nonce_store: NonceStore, *, skew_seconds: float = 300.0, now: datetime.datetime | None = None) -> str | None: 
119def validate_freshness(
120    event: dict[str, Any],
121    nonce_store: NonceStore,
122    *,
123    skew_seconds: float = DEFAULT_SKEW_SECONDS,
124    now: datetime | None = None,
125) -> str | None:
126    """Run the NEW-4 post-HMAC-verification checks on a parsed message.
127
128    MUST only be called after :func:`axiam_sdk.amqp._hmac.verify_hmac` has
129    already succeeded for this exact delivery — ``key_version``, ``nonce``,
130    and ``issued_at`` are part of the signed body, so an attacker cannot
131    forge or strip them without also forging the HMAC signature.
132
133    Returns ``None`` if the message passes every check. Otherwise returns
134    a short, non-sensitive rejection reason string (safe to put in a
135    security log line — it never contains the actual nonce, timestamp, or
136    signature values) that the caller should log before nacking without
137    requeue.
138    """
139    key_version = event.get("key_version")
140    if not isinstance(key_version, int) or isinstance(key_version, bool):
141        return "missing or invalid key_version"
142    if key_version < MIN_KEY_VERSION:
143        return "unsupported key_version (replay protection unavailable)"
144
145    issued_at_raw = event.get("issued_at")
146    if not isinstance(issued_at_raw, str):
147        return "missing or invalid issued_at"
148    try:
149        issued_at = _parse_rfc3339(issued_at_raw)
150    except ValueError:
151        return "unparseable issued_at"
152
153    reference = now if now is not None else datetime.now(timezone.utc)
154    delta_seconds = abs((reference - issued_at).total_seconds())
155    if delta_seconds > skew_seconds:
156        return "stale issued_at outside allowed clock-skew window"
157
158    nonce = event.get("nonce")
159    if not isinstance(nonce, str) or not nonce:
160        return "missing or invalid nonce"
161
162    if not nonce_store.check_and_record(nonce):
163        return "replayed nonce"
164
165    return None
